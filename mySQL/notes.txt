CREATE DATABASE practice;

USE practice;

SHOW TABLES;

CREATE TABLE users (
  ID INT NOT NULL AUTO_INCREMENT,
  userName varchar(50) UNIQUE NOT NULL,
  about varchar(250),
  age INTEGER,
  PRIMARY KEY (ID)
);

DESCRIBE users;

CREATE TABLE games (
  ID INT NOT NULL AUTO_INCREMENT,
  winner varchar(50),
  PRIMARY KEY (ID)
);
